# Check port:
netstat -aof | findstr 5432

#Create Network
docker create library-network

# PostgreSQL container wheels in version 17.9
docker  run --name librarydb -p 5432:5432 -d --network library-network -e POSTGRES_PASSWORD=postgres -e POSTGRES_USER=postgres -e POSTGRES_DB=library postgres:17.9

#Pgadmin4 wheels
docker run --name pgadmin4 -p 15432:80 -d --network library-network -e PGADMIN_DEFAULT_EMAIL=admin@admin.com -e PGADMIN_PASSWORD=admin dpage/pgadmin4:9.13
